# 结构体指针
> 通过指针来输出结构体

```
#include<iostream>
#include<string>
using namespace std;

struct Teacher {
    string name;
    int age;
};

int main () {
    // 定义结构体
    Teacher t = {"张三", 29};
    // 指针绑定结构体地址
    Teacher * p = &t;
    // 通过指针访问结构体内的值
    cout << "姓名：" << p->name << endl;
    cout << "年纪：" << p->age << endl;
    return 0;
}
```
